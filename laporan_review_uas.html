// Pointer sebagai penghubung node (syarat 3)
class MenuLinkedList {
private:
    MenuItem* head;  // pointer ke node pertama
    int size;

public:
    void insert(const MenuItem& item) {
        MenuItem* node = new MenuItem(...);  // alokasi dinamis
        if (!head) {
            head = node;
        } else {
            MenuItem* curr = head;             // traversal pakai pointer
            while (curr->next) curr = curr->next;
            curr->next = node;
        }
    }

    MenuItem* find(int id) const {        // return pointer
        MenuItem* curr = head;
        while (curr) {
            if (curr->id == id) return curr;
            curr = curr->next;
        }
        return nullptr;
    }

    std::vector<MenuItem*> getAll() const { // vector of pointers
        std::vector<MenuItem*> result;
        MenuItem* curr = head;
        while (curr) { result.push_back(curr); curr = curr->next; }
        return result;
    }

    // Destructor: hapus semua node dengan delete (syarat 3)
    ~MenuLinkedList() {
        MenuItem* curr = head;
        while (curr) { MenuItem* tmp = curr->next; delete curr; curr = tmp; }
    }
};

template<typename T>
T* linearSearch(std::vector<T*>& vec, std::function<bool(T*)> predicate) {
    auto it = std::find_if(vec.begin(), vec.end(), predicate);
    return (it != vec.end()) ? *it : nullptr;
}

// Template 2 — mencari SEMUA elemen yang cocok
template<typename T>
std::vector<T*> linearSearchAll(std::vector<T*>& vec, std::function<bool(T*)> predicate) {
    std::vector<T*> result;
    std::copy_if(vec.begin(), vec.end(), std::back_inserter(result), predicate);
    return result;
}

Linked list single untuk master data menu. Operasi: Insert (append ke tail), Remove (hapus by ID, handle kasus head & tengah), Find (traversal by ID), Display (getAll).

datastructures.h · Operasi Insert, Remove, Find, DisplayC++
class MenuLinkedList {
    MenuItem* head; int size;

    void insert(const MenuItem& item) {   // INSERT
        MenuItem* node = new MenuItem(...);
        if (!head) { head = node; }
        else { MenuItem* curr=head; while(curr->next) curr=curr->next; curr->next=node; }
        size++;
    }

    bool remove(int id) {              // REMOVE (hapus by id)
        if (!head) return false;
        if (head->id == id) { MenuItem* tmp=head; head=head->next; delete tmp; size--; return true; }
        MenuItem* curr=head;
        while(curr->next && curr->next->id!=id) curr=curr->next;
        if(!curr->next) return false;
        MenuItem* tmp=curr->next; curr->next=tmp->next; delete tmp; size--; return true;
    }

    MenuItem* find(int id) const {    // FIND
        MenuItem* curr=head;
        while(curr) { if(curr->id==id) return curr; curr=curr->next; }
        return nullptr;
    }

    std::vector<MenuItem*> getAll() const { // DISPLAY
        std::vector<MenuItem*> result;
        MenuItem* curr=head;
        while(curr) { result.push_back(curr); curr=curr->next; }
        return result;
    }
};

void insert(const Table& t) {   // INSERT — tail->next = head (circular)
    Table* node = new Table(t.tableNumber, t.capacity);
    if (!head) { head = tail = node; node->next = head; }
    else { tail->next = node; tail = node; tail->next = head; }
    size++;
}

Table* next() {               // ROTATE — geser ke meja berikutnya
    if (!current) return nullptr;
    current = current->next;   // karena circular, tidak pernah nullptr
    return current;
}

Table* findTable(int num) const {  // FIND di circular list
    if (!head) return nullptr;
    Table* curr = head;
    do {
        if (curr->tableNumber == num) return curr;
        curr = curr->next;
    } while (curr != head);   // berhenti saat kembali ke head
    return nullptr;
}

// Untuk staf — rotasi giliran bertugas
Staff* rotateShift() {
    if (!onDuty) return nullptr;
    onDuty->onDuty = false;
    onDuty = onDuty->next;      // muter ke staf berikutnya
    onDuty->onDuty = true;
    return onDuty;
}

    Action pop() {                 // POP (LIFO — ambil terakhir)
        if (history.empty())
            throw std::runtime_error("Tidak ada aksi untuk di-undo!");
        Action a = history.top();
        history.pop();
        return a;
    }

    const Action& peek() const {  // PEEK — lihat top tanpa hapus
        if (history.empty())
            throw std::runtime_error("Stack kosong!");
        return history.top();
    }
};
STRUKDAT 4 Priority Queue — Antrian Order VIP ✅ TERPENUHI (Priority Queue)
📄 datastructures.h · OrderComparator + class OrderQueue · Line 365–420

Implementasi Priority Queue menggunakan std::priority_queue dengan custom comparator. Priority 1 = VIP (diproses lebih dulu), Priority 2 = Normal. Dalam prioritas sama, berlaku FIFO berdasarkan orderId. Operasi: Enqueue, Dequeue (ambil prioritas tertinggi), Front.

datastructures.h · Priority Queue + ComparatorC++
// Custom comparator — priority kecil = lebih penting
struct OrderComparator {
    bool operator()(const Order* a, const Order* b) const {
        if (a->priority != b->priority)
            return a->priority > b->priority; // 1(VIP) < 2(Normal)
        return a->orderId > b->orderId;       // FIFO dalam prioritas sama
    }
};

class OrderQueue {
    std::priority_queue<Order*, std::vector<Order*>, OrderComparator> pq;
    std::list<Order*> allOrders;

    Order* enqueue(int tableNum, int priority = 2) {  // ENQUEUE
        Order* order = new Order(nextId++, tableNum, priority);
        pq.push(order);
        allOrders.push_back(order);
        return order;
    }

    Order* dequeue() {                              // DEQUEUE (ambil prioritas tertinggi)
        if (pq.empty()) throw std::runtime_error("Antrian order kosong!");
        Order* order = pq.top(); pq.pop();
        return order;
    }
};
STRUKDAT 5 AVL Tree — Pencarian Menu O(log n) ✅ TERPENUHI
📄 datastructures.h · AVLNode + class AVLTree · Line 425–530

AVL Tree lengkap dengan rotasi kiri/kanan dan balance factor. Digunakan untuk pencarian menu berdasarkan ID secara O(log n). Operasi: insert (dengan auto-rebalancing), remove, search, inorder traversal.

datastructures.h · AVL rotasi dan balanceC++
static AVLNode* rotateRight(AVLNode* y) {  // Rotasi kanan
    AVLNode* x = y->left; AVLNode* T2 = x->right;
    x->right = y; y->left = T2;
    updateHeight(y); updateHeight(x);
    return x;
}

static AVLNode* balance(AVLNode* n) {      // Auto-balancing
    updateHeight(n);
    int bf = balanceFactor(n);
    if (bf > 1) {   // Left heavy
        if (balanceFactor(n->left) < 0) n->left = rotateLeft(n->left);
        return rotateRight(n);
    }
    if (bf < -1) {  // Right heavy
        if (balanceFactor(n->right) > 0) n->right = rotateRight(n->right);
        return rotateLeft(n);
    }
    return n;
}

MenuItem* search(int id) const {           // SEARCH O(log n)
    AVLNode* curr = root;
    while (curr) {
        if (id == curr->data.id) return &curr->data;
        curr = (id < curr->data.id) ? curr->left : curr->right;
    }
    return nullptr;
}

class MenuRecommendationGraph {
    // Adjacency List: Item -> Daftar (Item Teman, Frekuensi Beli Bareng)
    std::unordered_map<std::string, std::unordered_map<std::string, int>> adjList;

    // [BFS] — Rekomendasi Level 1 (Ekstraksi Tetangga Langsung)
    std::vector<std::string> getRecommendationsBFS(const std::string& startItem) {
        std::queue<std::string> q;
        q.push(startItem);
        while (!q.empty()) {
            std::string curr = q.front(); q.pop();
            for (const auto& neighbor : adjList[curr]) {
                level1Recs.push_back(neighbor.first); // Ambil tetangga
            }
            break; // Berhenti di Level 1
        }
        return level1Recs;
    }

    // [DFS] — Pencarian Rantai Pesanan Kombo Berantai
    std::vector<std::string> getRecommendationComboDFS(const std::string& startItem) {
        std::stack<std::string> s;
        s.push(startItem);
        while (!s.empty()) {
            std::string curr = s.top(); s.pop();
            if (!visited[curr]) {
                visited[curr] = true;
                comboPath.push_back(curr);
                // Lanjut jelajahi node anak terdalam (DFS)
                for (const auto& neighbor : adjList[curr])
                    if (!visited[neighbor.first]) s.push(neighbor.first);
            }
        }
        return comboPath;
    }
};
STRUKDAT 7 Hash Table — Lookup Menu O(1) ✅ TERPENUHI
📄 datastructures.h · class MenuHashTable · Line 605–660

Hash table berbasis std::unordered_map dengan dua index: by ID (integer) dan by name (string lowercase). Pencarian O(1) average. Operasi: insert, remove, findById, findByName, countByCategory.

datastructures.h · MenuHashTableC++
class MenuHashTable {
    std::unordered_map<int, MenuItem>        table;      // hash by id
    std::unordered_map<std::string, int>  nameIndex;  // hash by name

    void insert(const MenuItem& item) {
        table[item.id] = item;                        // O(1) insert
        nameIndex[toLower(item.name)] = item.id;
    }

    MenuItem* findById(int id) {          // O(1) lookup by ID
        auto it = table.find(id);
        return (it != table.end()) ? &it->second : nullptr;
    }

    MenuItem* findByName(const std::string& name) { // O(1) lookup by name
        auto it = nameIndex.find(toLower(name));
        if (it == nameIndex.end()) return nullptr;
        return findById(it->second);
    }

    int countByCategory(const std::string& cat) const {
        return (int)std::count_if(table.begin(), table.end(),
            [&cat](const auto& pair) { return pair.second.category == cat; });
    }
};

namespace Sorting {

    // 1. Bubble Sort — menu berdasarkan harga (manual, O(n²))
    inline void bubbleSortByPrice(std::vector<MenuItem>& vec, bool ascending = true) {
        int n = vec.size();
        for (int i = 0; i < n-1; i++)
            for (int j = 0; j < n-i-1; j++)
                if ((ascending && vec[j].price > vec[j+1].price) ||
                    (!ascending && vec[j].price < vec[j+1].price))
                    std::swap(vec[j], vec[j+1]);
    }

    // 2. STL sort + lambda — menu berdasarkan nama (O(n log n))
    inline void sortByName(std::vector<MenuItem>& vec) {
        std::sort(vec.begin(), vec.end(),
            [](const MenuItem& a, const MenuItem& b) { return a.name < b.name; });
    }

    // 3. Merge Sort — order berdasarkan total harga (O(n log n))
    inline void merge(std::vector<Order*>& vec, int l, int m, int r) {
        std::vector<Order*> left(vec.begin()+l, vec.begin()+m+1);
        std::vector<Order*> right(vec.begin()+m+1, vec.begin()+r+1);
        int i=0, j=0, k=l;
        while (i<(int)left.size() && j<(int)right.size())
            vec[k++] = (left[i]->totalPrice >= right[j]->totalPrice) ? left[i++] : right[j++];
        while (i<(int)left.size()) vec[k++]=left[i++];
        while (j<(int)right.size()) vec[k++]=right[j++];
    }

    inline void mergeSort(std::vector<Order*>& vec, int l, int r) {
        if (l >= r) return;
        int m = (l + r) / 2;
        mergeSort(vec, l, m); mergeSort(vec, m+1, r);
        merge(vec, l, m, r);
    }
}
